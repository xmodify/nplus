    report_date,
    shift_time,
    nurse_fulltime,
    nurse_partime,
    nurse_oncall,
    recorder,
    note,
    patient_all,
    patient_critical,
    patient_semi_critical,
    patient_moderate,
    patient_convalescent,
    nursing_hours,
    working_hours,
    nhppd,
    nurse_shift_time,
    productivity,
    created_at,
    updated_at
)
SELECT
    report_date,
    shift_time,
    nurse_fulltime,
    nurse_partime,
    nurse_oncall,
    recorder,
    note,
    patient_all,
    critical_ill        AS patient_critical,
    semi_critical_ill   AS patient_semi_critical,
    moderate_ill        AS patient_moderate,
    convalescent        AS patient_convalescent,
    patient_hr          AS nursing_hours,
    nurse_hr            AS working_hours,
    hhpuos              AS nhppd,

    created_at,
