INSERT INTO productivity_er (
    report_date,
    shift_time,
    nurse_fulltime,
    nurse_partime,
    nurse_oncall,
    recorder,
    note,
    patient_all,
    patient_resuscitation,
    patient_emergent,
    patient_urgent,
    patient_semi_urgent,
    patient_non_urgent,
    nursing_hours,
    working_hours,
    nhppd,
    nurse_shift_time,
    productivity,
    created_at,
    updated_at
)
SELECT
    report_date,
    shift_time,
    nurse_fulltime,
    nurse_partime,
    nurse_oncall,
    recorder,
    note,
    patient_all,
    acute_illness        AS patient_resuscitation,
    emergent             AS patient_emergent,
    urgent               AS patient_urgent,
    0                    AS patient_semi_urgent,
    non_acute_illness    AS patient_non_urgent,
    patient_hr           AS nursing_hours,
    nurse_hr             AS working_hours,
    hhpuos               AS nhppd,
    nurse_shift_time,
    productivity,
    created_at,
    updated_at
FROM nurse_productivity_ers;
